tabela <- Matriz〈não-terminais, terminais, produção da gramática〉
para cada item f de first:
    esquerda <- produção correspondente a f
    se ε pertence a f:
        para cada símbolo b em follow(lado esquerdo da produção esquerda):
            tabela[lado esquerdo da produção esquerda][b] <- esquerda
    para cada símbolo b em f:
        se b não é igual a ε:
            tabela[lado esquerdo da produção esquerda][b] <- esquerda
